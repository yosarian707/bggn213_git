generate_dna <- function(length){
  bases <- c("A", "T", "G", "C")
sequence <- sample(bases, size=length, replace=TRUE)
return(sequence)
}

funcy <- function(pID){
  library(bio3d)
  s <- read.pdb(pID)
  s.ChainA <- trim.pdb(s, chain ="A", elety = "CA")
  s.b <- s.ChainA$atom$b
  plotb3(s.b, sse=s.ChainA, typ="l", ylab="Bfactor")
}
